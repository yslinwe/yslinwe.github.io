	void replaceSpace(char* str, int length) {
		int count = 0;
		for (int i = 0; i < length; i++)
		{
			if (str[i] == ' ')
				count++;
		}
		int new_length = length + 2 * count;
		int j = length-1;
		for (int i = new_length-1; i >= 0;)
		{
			if (str[j] == ' ')
			{
				str[i--] = '0';
				str[i--] = '2';
				str[i--] = '%';
			}
			else
			{
				str[i--] = str[j];
			}
			j--;
		}
	}

	void replaceSpace2(char* str, int length) {
		int i = 0;
		int numSpace = 0;
		while (str[i] != '\0')
		{
			if (str[i] == ' ')
				numSpace++;
			++i;
		}
		int newLen = i + numSpace * 2;
		if (newLen > length)
			return;
		for (int j = i; j >= 0, newLen >= 0;)
		{
			if (str[j] == ' ')
			{
				str[newLen--] = '0';
				str[newLen--] = '2';
				str[newLen--] = '%';
			}
			else
				str[newLen--] = str[j];
			j--;
		}
	}
