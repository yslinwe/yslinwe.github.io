### 题目描述

输入一个链表，输出该链表中倒数第k个结点。

如果该链表长度小于k，请返回空。 

### 示例1

#### 输入

```
{1,2,3,4,5},1 
```

#### 返回值

```
{5}
```

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 *	ListNode(int x) : val(x), next(nullptr) {}
 * };
 */
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pHead ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* FindKthToTail(ListNode* pHead, int k) {
        // write code here
      
        ListNode *result = pHead;
        while(pHead&&k--)
        {
            pHead = pHead->next;
        }
        if(k>0)
           return NULL;
        while(pHead)
        {
            pHead = pHead->next;
            result = result->next;
        }
        return result;
           
    }
};
```
