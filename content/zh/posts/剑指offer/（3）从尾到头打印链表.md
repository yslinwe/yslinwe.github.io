### 题目描述

输入一个链表，按链表从尾到头的顺序返回一个ArrayList。

### 示例1

#### 输入

```
{67,0,24,58}
```

#### 返回值

```
[58,24,0,67]
```

```C++
/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        stack<int> temp;
        while(head!=NULL)
        {
            temp.push(head->val);
            head = head->next;
        }
        vector<int> result;
        int tempS = temp.size();
        for(int i = 0;i<tempS;i++)
        {
            result.push_back(temp.top());
            temp.pop();
        }
        return result;
    }
};
```
