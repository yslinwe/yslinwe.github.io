/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        stack<int> temp;
        while(head!=NULL)
        {
            temp.push(head->val);
            head = head->next;
        }
        vector<int> result;
        int tempS = temp.size();
        for(int i = 0;i<tempS;i++)
        {
            result.push_back(temp.top());
            temp.pop();
        }
        return result;
    }
};
