class Solution {

        if(numbers.empty())
            return 0;
        int time = 0;
        int result = 0;
        for(int i=0;i<numbers.size();i++)
        {
            if(time==0)
            {
                result = numbers[i];
                time = 1;
            }
            else if(numbers[i]==result)
                time++;
            else
                time--;
        }
        time = 0;
        for(int i = 0;i<numbers.size();i++)
            if(numbers[i]==result)
                time++;
        return time>numbers.size()>>1?result:0;
    }
};
