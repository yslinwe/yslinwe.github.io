class Solution {
public:
    bool IsPopOrder(vector<int> pushV,vector<int> popV) {
        bool result = false;
        stack<int> leftS;
        stack<int> rightS;
        int pushVSize = pushV.size();
        int popVSize = popV.size();
        int j = 0;
        bool isLeft = true;
        for(int i = 0;i<pushVSize;i++)
        {
            if(pushV[i] == popV[j])
            {
                j++;
                isLeft = !isLeft;
            }else
            {
                if(isLeft)
                    leftS.push(pushV[i]);
                else
                    rightS.push(pushV[i]);
            }
        }
        if(j>=popVSize)
            return true;
       int left = 0;
        int right = 0;
        while(j<popVSize)
        {
            left = leftS.empty()?NULL:leftS.top();
            right = rightS.empty()?NULL:rightS.top();
            
            if(popV[j]==left)
            {
                 leftS.pop();
                 result = true;
            }
            else if(popV[j]==right)
            {
                rightS.pop();
                result = true;
            }
            else
            {
                result = false;
                break;
            }
            j++;
        }
        return result;
    }
};
