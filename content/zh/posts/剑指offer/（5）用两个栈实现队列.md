### 题目描述

用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。

```c++
class Solution
{
public:
    void push(int node) {
        stack1.push(node);
    }

    int pop() {
        if(stack2.empty()&&!stack1.empty())
        {
            int stackS = stack1.size();
            for(int i =0;i<stackS;i++)
            {
                int temp = stack1.top();
                stack1.pop();
                stack2.push(temp);
            }
        }
        int val = stack2.top();
        stack2.pop();
        return val;
    }

private:
    stack<int> stack1;
    stack<int> stack2;
};
```
